001package com.hypernite.mc.hnmc.core.managers.builder;
002
003import net.md_5.bungee.api.chat.BaseComponent;
004import org.bukkit.entity.Entity;
005import org.bukkit.entity.Player;
006import org.bukkit.inventory.ItemStack;
007
008import java.util.function.Consumer;
009
010public interface AbstractMessageBuilder extends Buildable<BaseComponent[]> {
011
012    /**
013     * @param msg 訊息串
014     * @return this
015     */
016    AbstractMessageBuilder add(String... msg);
017
018    /**
019     * @param website 可點擊連結
020     * @return this
021     */
022    AbstractMessageBuilder url(String website);
023
024    /**
025     * @param command 指令輸入
026     * @return this
027     */
028    AbstractMessageBuilder suggest(String command);
029
030    /**
031     * @param command 指令執行
032     * @return this
033     */
034    AbstractMessageBuilder command(String command);
035
036    /**
037     * @param page 頁數
038     * @return this
039     */
040    AbstractMessageBuilder page(String page);
041
042    /**
043     * @param texts 文字氣泡
044     * @return this
045     */
046    AbstractMessageBuilder hoverText(String... texts);
047
048    /**
049     * 顯示物品資訊
050     *
051     * @param item 物品
052     * @return this
053     */
054    AbstractMessageBuilder showItem(ItemStack item);
055
056    /**
057     * 顯示實體資訊
058     *
059     * @param entity 實體
060     * @return this
061     */
062    AbstractMessageBuilder showEntity(Entity entity);
063
064    /**
065     * 顯示成就資訊
066     *
067     * @param achievementNode 成就節點
068     * @return this
069     */
070    AbstractMessageBuilder showAdvancement(String achievementNode);
071
072    /**
073     * @param insert 訊息
074     * @return this
075     */
076    AbstractMessageBuilder insertWhenShiftClick(String insert);
077
078    /**
079     * 默認為 十分鐘 之後自動過期
080     *
081     * @param runner 運行函式
082     * @return this
083     */
084    AbstractMessageBuilder run(Consumer<Player> runner);
085
086    /**
087     * @param timeoutClicks 點擊多少次後自動過期
088     * @param runner        運行函式
089     * @return this
090     */
091    AbstractMessageBuilder runClicks(int timeoutClicks, Consumer<Player> runner);
092
093    /**
094     * @param timeoutSeconds 多少秒後失效
095     * @param runner         運行函式
096     * @return this
097     */
098    AbstractMessageBuilder runTimeout(int timeoutSeconds, Consumer<Player> runner);
099
100    /**
101     * 隔行
102     *
103     * @return this
104     */
105    AbstractMessageBuilder nextLine();
106
107    void sendPlayer(Player player);
108}
