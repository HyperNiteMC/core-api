001package com.hypernite.mc.hnmc.core.misc.permission;
002
003import org.bukkit.command.CommandSender;
004
005import javax.annotation.Nonnull;
006import java.util.ArrayList;
007import java.util.List;
008
009/**
010 * 權限 API
011 */
012public class Perm {
013    public static final String ADMIN = "hypernite.admin";
014    public static final String HELPER = "hypernite.helper";
015    public static final String MOD = "hypernite.mod";
016    public static final String BUILDER = "hypernite.builder";
017    public static final String DONOR = "hypernite.donor";
018    public static final String DEVELOPER = "hypernite.developer";
019    public static final String OWNER = "*";
020
021    public static boolean hasPermission(CommandSender sender, String permission) {
022        boolean reuslt = false;
023        if (permission == null) return true;
024        for (String Pnode : getPermHierarchy(permission)) {
025            reuslt = reuslt || sender.hasPermission(Pnode);
026        }
027        return reuslt;
028    }
029
030    public static List<String> getPermHierarchy(@Nonnull String string) {
031        String[] node = string.split("\\.");
032        List<String> permissionNode = new ArrayList<>();
033        StringBuilder stringBuilder = new StringBuilder();
034        for (int i = 0; i < node.length; i++) {
035            stringBuilder.append(node[i]);
036            permissionNode.add(stringBuilder.toString());
037            if (i != node.length - 1) stringBuilder.append(".");
038        }
039        return permissionNode;
040    }
041}
