001package com.hypernite.mc.hnmc.core.misc.commands;
002
003
004import com.hypernite.mc.hnmc.core.misc.commands.exception.CommandArgumentException;
005import com.hypernite.mc.hnmc.core.misc.commands.exception.CommandPermissionException;
006import com.hypernite.mc.hnmc.core.misc.permission.Perm;
007import org.bukkit.command.CommandSender;
008
009import javax.annotation.Nonnull;
010import java.util.ArrayList;
011import java.util.Arrays;
012import java.util.List;
013
014public abstract class CommandNode {
015
016    private final String description;
017    private final String placeholder;
018    private final String command;
019
020
021    private final ArrayList<CommandNode> subCommands = new ArrayList<>();
022
023
024    private final ArrayList<String> alias = new ArrayList<>();
025
026
027    private final String permission;
028
029
030    private CommandNode parent;
031
032
033    /**
034     * @param parent      父類指令節點
035     * @param command     指令
036     * @param permission  權限
037     * @param description 介紹
038     * @param placeholder 佔位符
039     * @param alias       縮寫指令
040     */
041    public CommandNode(CommandNode parent, @Nonnull String command, String permission, @Nonnull String description, String placeholder, String... alias) {
042        this.command = command;
043        this.parent = parent;
044        this.alias.add(command);
045        this.permission = permission;
046        this.description = description;
047        this.placeholder = placeholder;
048        this.alias.addAll(List.of(alias));
049    }
050
051    public String getDescription() {
052        return description;
053    }
054
055    public String getPlaceholder() {
056        return placeholder;
057    }
058
059    public ArrayList<CommandNode> getSubCommands() {
060        return subCommands;
061    }
062
063    public ArrayList<String> getAlias() {
064        return alias;
065    }
066
067    public CommandNode getParent() {
068        return parent;
069    }
070
071    void setParent(CommandNode node) {
072        this.parent = node;
073    }
074
075    public String getPermission() {
076        return permission;
077    }
078
079    public String getCommand() {
080        return command;
081    }
082
083    @Deprecated
084    public void addAlias(@Nonnull String ali) {
085        if (!alias.contains(ali)) {
086            this.alias.add(ali);
087        }
088    }
089
090    public void addAllAliases(@Nonnull List<String> aliases) {
091        aliases.forEach(ali -> {
092            if (!alias.contains(ali)) {
093                alias.add(ali);
094            }
095        });
096    }
097
098    public void addSub(@Nonnull CommandNode c) {
099        subCommands.add(c);
100    }
101
102    /**
103     * @param sender 指令發送者
104     * @param args   指令參數
105     * @return 執行是否成功
106     */
107    public abstract boolean executeCommand(@Nonnull CommandSender sender, @Nonnull List<String> args);
108
109    /**
110     * @param sender 指令放著
111     * @param args   指令參數
112     * @return Tab 列表
113     */
114    public abstract List<String> executeTabCompletion(@Nonnull CommandSender sender, @Nonnull List<String> args);
115
116    public boolean invokeCommand(@Nonnull CommandSender sender, @Nonnull List<String> args) throws CommandPermissionException, CommandArgumentException {
117
118        if (permission != null && !Perm.hasPermission(sender, permission)) {
119            throw new CommandPermissionException(permission);
120        }
121
122        if (args.size() > 0) {
123            for (CommandNode subCommand : subCommands) {
124                if (subCommand.match(args.get(0))) {
125                    List<String> passArg = new ArrayList<>(args);
126                    passArg.remove(0);
127                    if (subCommand.getPlaceholder() != null) {
128                        String[] placeholders = Arrays.stream(subCommand.getPlaceholder().split(" ")).filter(holder -> holder.startsWith("<") && holder.endsWith(">")).toArray(String[]::new);
129                        if (passArg.size() < placeholders.length) {
130                            throw new CommandArgumentException(String.join(" ", placeholders));
131                        }
132                    }
133
134                    return subCommand.invokeCommand(sender, passArg);
135                }
136            }
137        }
138
139        if (this.getPlaceholder() != null) {
140            String[] placeholders = Arrays.stream(this.getPlaceholder().split(" ")).filter(holder -> holder.startsWith("<") && holder.endsWith(">")).toArray(String[]::new);
141            if (args.size() < placeholders.length) {
142                throw new CommandArgumentException(String.join(" ", placeholders));
143            }
144        }
145        return executeCommand(sender, args);
146    }
147
148    public List<String> invokeTabCompletion(@Nonnull CommandSender sender, @Nonnull List<String> args) throws CommandPermissionException {
149        if (permission != null && !Perm.hasPermission(sender, permission)) {
150            throw new CommandPermissionException(permission);
151        }
152
153        if (args.size() > 0) {
154            for (CommandNode subCommand : subCommands) {
155                if (subCommand.match(args.get(0))) {
156                    List<String> passArg = new ArrayList<>(args);
157                    passArg.remove(0);
158                    return subCommand.invokeTabCompletion(sender, passArg);
159                }
160            }
161        }
162
163
164        return executeTabCompletion(sender, args);
165    }
166
167    public boolean match(@Nonnull String args) {
168        for (String ali : alias) {
169            if (args.equalsIgnoreCase(ali)) {
170                return true;
171            }
172        }
173        return false;
174    }
175}
