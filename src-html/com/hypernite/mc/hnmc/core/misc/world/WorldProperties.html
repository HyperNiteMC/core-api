001package com.hypernite.mc.hnmc.core.misc.world;
002
003import org.bukkit.Location;
004
005public class WorldProperties {
006
007    private boolean pvp;
008    private boolean pve;
009    private Location spawn;
010    private boolean vulnerable;
011    private boolean autoLoad;
012    private WorldProperties(WorldProperties properties) {
013        this(properties.pvp, properties.pve, properties.getSpawn(), properties.vulnerable, properties.autoLoad);
014    }
015    public WorldProperties(boolean pvp, boolean pve, Location spawn, boolean vulnerable, boolean autoLoad) {
016        this.pvp = pvp;
017        this.pve = pve;
018        this.spawn = spawn;
019        this.vulnerable = vulnerable;
020        this.autoLoad = autoLoad;
021    }
022
023    public WorldProperties(Location spawn) {
024        this.spawn = spawn;
025    }
026
027    public static WorldProperties copyOf(WorldProperties properties) {
028        return new WorldProperties(properties);
029    }
030
031    public boolean isPvp() {
032        return pvp;
033    }
034
035    public void setPvp(boolean pvp) {
036        this.pvp = pvp;
037    }
038
039    public boolean isPve() {
040        return pve;
041    }
042
043    public void setPve(boolean pve) {
044        this.pve = pve;
045    }
046
047    public Location getSpawn() {
048        return spawn;
049    }
050
051    public void setSpawn(Location spawn) {
052        this.spawn = spawn;
053    }
054
055    public boolean isVulnerable() {
056        return vulnerable;
057    }
058
059    public void setVulnerable(boolean vulnerable) {
060        this.vulnerable = vulnerable;
061    }
062
063    public boolean isAutoLoad() {
064        return autoLoad;
065    }
066
067    public void setAutoLoad(boolean autoLoad) {
068        this.autoLoad = autoLoad;
069    }
070}
