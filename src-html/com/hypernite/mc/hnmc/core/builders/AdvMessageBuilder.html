001package com.hypernite.mc.hnmc.core.builders;
002
003import com.hypernite.mc.hnmc.core.main.HyperNiteMC;
004import com.hypernite.mc.hnmc.core.managers.builder.AbstractAdvMessageBuilder;
005import com.hypernite.mc.hnmc.core.managers.builder.AbstractMessageBuilder;
006import net.md_5.bungee.api.chat.BaseComponent;
007import net.md_5.bungee.api.chat.TextComponent;
008import org.bukkit.entity.Player;
009
010/**
011 * @see MessageBuilder
012 */
013public class AdvMessageBuilder implements AbstractAdvMessageBuilder {
014
015    private final AbstractAdvMessageBuilder delegate;
016
017    public AdvMessageBuilder(String... msg) {
018        this.delegate = HyperNiteMC.getAPI().getFactory().getBuilder().getAdvMessageBuilder(msg);
019    }
020
021    @Override
022    public AbstractAdvMessageBuilder add(String... msg) {
023        return delegate.add(msg);
024    }
025
026    @Override
027    public AbstractAdvMessageBuilder add(BaseComponent... baseComponent) {
028        return delegate.add(baseComponent);
029    }
030
031    @Override
032    public AbstractAdvMessageBuilder add(AbstractMessageBuilder... builders) {
033        return delegate.add(builders);
034    }
035
036    @Override
037    public AbstractAdvMessageBuilder nextLine() {
038        return delegate.nextLine();
039    }
040
041    @Override
042    public void sendPlayer(Player player) {
043        delegate.sendPlayer(player);
044    }
045
046    @Override
047    public TextComponent build() {
048        return delegate.build();
049    }
050}
