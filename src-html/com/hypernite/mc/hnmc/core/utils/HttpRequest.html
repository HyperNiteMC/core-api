001package com.hypernite.mc.hnmc.core.utils;
002
003import java.io.BufferedReader;
004import java.io.IOException;
005import java.io.InputStreamReader;
006import java.net.HttpURLConnection;
007import java.net.URL;
008import java.util.concurrent.CompletableFuture;
009import java.util.concurrent.CompletionException;
010
011/**
012 * HttpRequest 工具
013 */
014public class HttpRequest {
015
016    public static CompletableFuture<String> getFuture(String link) {
017        return CompletableFuture.supplyAsync(() -> {
018            try {
019                return get(link);
020            } catch (IOException e) {
021                throw new CompletionException(e);
022            }
023        });
024    }
025
026    public static String get(String link) throws IOException {
027        URL url = new URL(link);
028        HttpURLConnection con = (HttpURLConnection) url.openConnection();
029        con.setRequestMethod("GET");
030        con.setRequestProperty("User-Agent", "Mozilla/5.0");
031        BufferedReader in = new BufferedReader(
032                new InputStreamReader(con.getInputStream()));
033        String inputLine;
034        StringBuilder response = new StringBuilder();
035        while ((inputLine = in.readLine()) != null) {
036            response.append(inputLine);
037        }
038        in.close();
039        return response.toString();
040    }
041}
