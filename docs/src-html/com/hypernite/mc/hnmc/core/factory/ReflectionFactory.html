001package com.hypernite.mc.hnmc.core.factory;
002
003import java.util.Optional;
004
005public interface ReflectionFactory {
006
007    /**
008     * @param constructorParams 建構子參數
009     * @return this
010     */
011    ReflectionFactory setConstructor(Object... constructorParams);
012
013    /**
014     * @param methodName 方法名稱
015     * @param parameters 參數
016     * @return this
017     */
018    ReflectionFactory addMethod(String methodName, Object... parameters);
019
020    /**
021     * @param methodName static 方法名稱
022     * @param parameters 參數
023     * @return this
024     */
025    ReflectionFactory addStaticMethod(String methodName, Object... parameters);
026
027    /**
028     * @return Optional MethodWrapper
029     */
030    Optional<MethodWrapper> getMethods();
031
032    /**
033     * @param exception Exception class
034     * @param <E>       Exception Type
035     * @return Optional MethodWrapper
036     * @throws E Throwable Type
037     */
038    <E extends Throwable> Optional<MethodWrapper> getMethodsWithException(Class<E> exception) throws E;
039
040    /**
041     * @return Optional MethodWrapper
042     */
043    Optional<MethodWrapper> getStaticMethods();
044
045    /**
046     * @return MethodWrapper
047     * @throws ClassNotFoundException 找不到 class
048     */
049    MethodWrapper getStaticMethodsWithException() throws ClassNotFoundException;
050
051}
