001package com.hypernite.mc.hnmc.core.managers.builder;
002
003import org.bukkit.Material;
004import org.bukkit.inventory.ItemStack;
005
006public interface Builder {
007
008    /**
009     * @param msg 訊息串
010     * @return 訊息建造器
011     */

014    /**
015     * @param row   此背包界面最大行數
016     * @param title 標題
017     * @return 界面建造器
018     */
019    AbstractInventoryBuilder getInventoryBuilder(int row, String title);
020
021    /**
022     * @param msg 原始訊息串
023     * @return 訊息建造器
024     */
025    AbstractMessageBuilder getMessageBuilder(String... msg);
026
027    /**
028     * @return 物品建造器
029     */
030    AbstractItemStackBuilder getItemStackBuilder();
031
032    /**
033     * @param material 物品類型
034     * @return 物品建造器
035     */
036    AbstractItemStackBuilder getItemStackBuilder(Material material);
037
038    /**
039     * @param item 原始物品
040     * @return 物品建造器
041     */
042    AbstractItemStackBuilder getItemStackBuilder(ItemStack item);
043
044}
