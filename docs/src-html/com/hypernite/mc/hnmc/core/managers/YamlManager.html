001package com.hypernite.mc.hnmc.core.managers;
002
003import com.hypernite.mc.hnmc.core.config.yaml.Configuration;
004
005/**
006 * 新的 Config 管理器
007 */
008public interface YamlManager {
009
010    /**
011     * 重載所有文件
012     *
013     * @return 成功
014     */
015    boolean reloadConfigs();
016
017
018    /**
019     *
020     * @param yml 文件名稱
021     * @param <T> 映射接口
022     * @return 映射物件
023     */
024    <T extends Configuration> T getConfig(String yml);
025
026    /**
027     *
028     * @param config 映射物件類
029     * @param <T> 映射接口
030     * @return 映射物件
031     */
032    <T extends Configuration> T getConfigAs(Class<T> config);
033}
