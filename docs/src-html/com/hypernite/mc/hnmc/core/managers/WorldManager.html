001package com.hypernite.mc.hnmc.core.managers;
002
003import com.hypernite.mc.hnmc.core.misc.world.WorldNonExistException;
004import com.hypernite.mc.hnmc.core.misc.world.WorldProperties;
005import org.bukkit.GameRule;
006import org.bukkit.Location;
007import org.bukkit.World;
008import org.bukkit.WorldType;
009import org.bukkit.command.CommandSender;
010import org.bukkit.configuration.file.FileConfiguration;
011import org.bukkit.generator.ChunkGenerator;
012
013import javax.annotation.Nonnull;
014import java.util.List;
015import java.util.Map;
016import java.util.Optional;
017import java.util.concurrent.CompletableFuture;
018
019/**
020 * 用於取代 Multiverse-Core
021 */
022public interface WorldManager {
023
024    /**
025     * @param name Bukkit世界
026     * @return 世界設定
027     * @throws WorldNonExistException 世界不存在
028     */
029    WorldProperties getWorldProperties(String name) throws WorldNonExistException;
030
031    /**
032     * @param world Bukkit世界
033     * @return 世界設定
034     */
035    WorldProperties getWorldProperties(World world);
036
037    /**
038     * @param name 世界名稱
039     * @return yaml 資料
040     * @throws WorldNonExistException 找不到世界
041     */
042    FileConfiguration getWorldFile(String name) throws WorldNonExistException;
043
044    /**
045     * @param world       世界
046     * @param type        類型
047     * @param environment 環境
048     * @return 加載是否成功
049     */
050    CompletableFuture<Boolean> futureCreateWorld(@Nonnull String world, WorldType type, World.Environment environment);
051
052    /**
053     * @param world              世界
054     * @param type               類型
055     * @param environment        環境
056     * @param generateStructures 生成建築
057     * @return 加載是否成功
058     */
059    CompletableFuture<Boolean> futureCreateWorld(@Nonnull String world, WorldType type, World.Environment environment, boolean generateStructures);
060
061    /**
062     * @param world              世界
063     * @param environment        環境
064     * @param generator          Chunk 生成器
065     * @param generateStructures 生成建築
066     * @param type               地形
067     * @param seed               種子碼
068     * @return 加載是否成功
069     */
070    CompletableFuture<Boolean> futureCreateWorld(@Nonnull String world, World.Environment environment, ChunkGenerator generator, boolean generateStructures, WorldType type, long seed);
071
072    /**
073     * @param world 世界名稱
074     * @return 加載是否成功
075     */
076    CompletableFuture<Boolean> futureLoadWorld(@Nonnull String world);
077
078    /**
079     * @param world  世界
080     * @param sender 指令發送者
081     * @throws WorldNonExistException 世界不存在
082     */
083    void loadWorld(@Nonnull String world, CommandSender sender) throws WorldNonExistException;
084
085    /**
086     * 保存所有世界設定
087     */
088    void saveAll();
089
090    /**
091     * @param world 世界名稱
092     * @return 是否成功
093     */
094    boolean unloadWorld(@Nonnull String world);
095
096    /**
097     * @param world 世界名稱
098     * @return 是否成功
099     * @throws WorldNonExistException 世界不存在
100     */
101    CompletableFuture<Boolean> deleteWorld(String world) throws WorldNonExistException;
102
103    /**
104     * @param world 世界名稱
105     * @return 出生點
106     */
107    Optional<Location> getWorldSpawn(String world);
108
109    /**
110     * @param world 世界名稱
111     * @return 是否成功
112     * @throws WorldNonExistException 世界不存在
113     */
114    boolean disableWorld(String world) throws WorldNonExistException;
115
116    /**
117     * @param world 世界
118     * @return 是否被禁用
119     */
120    boolean isNotAutoLoad(String world);
121
122    /**
123     * @param world  實際
124     * @param sender 指令發送者
125     * @throws WorldNonExistException 世界不存在
126     */
127    void enableWorld(String world, CommandSender sender) throws WorldNonExistException;
128
129    /**
130     * @return 帶有前綴和顏色辨別的世界列表
131     */
132    String[] listWorlds();
133
134    /**
135     * @return 純世界列表
136     */
137    List<String> listRealWorlds();
138
139    /**
140     * @param world 世界名稱
141     * @param name  Chunk 生成器 Id
142     * @return Chunk 生成器
143     */
144    Optional<ChunkGenerator> getChunkGenerator(String world, String name);
145
146    /**
147     * @param rules 遊戲規則
148     * @param world 世界
149     * @param <T>   設置類型, boolean 和 int 等等
150     */
151    <T> void applyGameRules(Map<GameRule<T>, T> rules, @Nonnull World world);
152
153}
