001package com.hypernite.mc.hnmc.core.managers;
002
003import com.hypernite.mc.hnmc.core.misc.world.WorldExistException;
004import com.hypernite.mc.hnmc.core.misc.world.WorldLoadedException;
005import com.hypernite.mc.hnmc.core.misc.world.WorldNonExistException;
006import com.hypernite.mc.hnmc.core.misc.world.WorldProperties;
007import org.bukkit.GameRule;
008import org.bukkit.World;
009import org.bukkit.WorldType;
010import org.bukkit.generator.ChunkGenerator;
011
012import javax.annotation.Nonnull;
013import java.util.Map;
014import java.util.Optional;
015import java.util.concurrent.CompletableFuture;
016import java.util.function.Consumer;
017
018/**
019 * 用於取代 Multiverse-Core
020 */
021public interface WorldManager {
022
023    /**
024     * @param name Bukkit世界
025     * @return 世界設定 (唯獨版本, 欲更新請使用 {@link WorldManager#updateWorldProperties(String, Consumer)})
026     * @throws WorldNonExistException 世界不存在
027     */
028    WorldProperties getWorldProperties(@Nonnull String name) throws WorldNonExistException;
029
030
031    boolean updateWorldProperties(@Nonnull String name, Consumer<WorldProperties> editor) throws WorldNonExistException;
032
033
034    /**
035     * @param world       世界
036     * @param type        類型
037     * @param environment 環境
038     * @return 加載是否成功
039     */
040    CompletableFuture<Boolean> createWorld(@Nonnull String world, WorldType type, World.Environment environment) throws WorldExistException;
041
042    /**
043     * @param world              世界
044     * @param type               類型
045     * @param environment        環境
046     * @param generateStructures 生成建築
047     * @return 加載是否成功
048     */
049    CompletableFuture<Boolean> createWorld(@Nonnull String world, WorldType type, World.Environment environment, boolean generateStructures) throws WorldExistException;
050
051    /**
052     * @param world              世界
053     * @param environment        環境
054     * @param generator          Chunk 生成器
055     * @param generateStructures 生成建築
056     * @param type               地形
057     * @param seed               種子碼
058     * @return 加載是否成功
059     */
060    CompletableFuture<Boolean> createWorld(@Nonnull String world, World.Environment environment, ChunkGenerator generator, boolean generateStructures, WorldType type, long seed) throws WorldExistException;
061
062    /**
063     * @param world 世界名稱
064     * @return 是否成功
065     * @throws WorldNonExistException 世界不存在
066     */
067    boolean deleteWorld(String world) throws WorldNonExistException;
068
069    /**
070     * @param world 世界名稱
071     * @return 加載是否成功
072     * @throws WorldNonExistException 世界不存在
073     */
074    CompletableFuture<Boolean> loadWorld(@Nonnull String world) throws WorldNonExistException, WorldLoadedException;
075
076    /**
077     * @param world 世界名稱
078     * @return 是否成功
079     * @throws WorldNonExistException 世界不存在
080     */
081    boolean unloadWorld(@Nonnull String world) throws WorldNonExistException;
082
083    /**
084     * @param world 世界名稱
085     * @return 是否成功
086     * @throws WorldNonExistException 世界不存在
087     */
088    boolean disableWorld(@Nonnull String world) throws WorldNonExistException;
089
090    /**
091     * @param world 實際
092     * @return 加載是否成功
093     * @throws WorldNonExistException 世界不存在
094     */
095    CompletableFuture<Boolean> enableWorld(@Nonnull String world) throws WorldNonExistException, WorldLoadedException;
096
097    /**
098     * 保存所有世界設定
099     */
100    void saveAll();
101
102    /**
103     * @return 帶有前綴和顏色辨別的世界列表
104     */
105    String[] listWorldMessages();
106
107    /**
108     * @return 純世界列表 和 啟用狀態。
109     */
110    Map<String, Boolean> getWorldList();
111
112    /**
113     * @param world 世界名稱
114     * @param name  Chunk 生成器 Id
115     * @return Chunk 生成器
116     */
117    Optional<ChunkGenerator> getChunkGenerator(@Nonnull String world, String name);
118
119    /**
120     * @param rules 遊戲規則
121     * @param world 世界
122     * @param <T>   設置類型, boolean 和 int 等等
123     */
124    <T> void applyGameRules(Map<GameRule<T>, T> rules, @Nonnull World world);
125
126}
