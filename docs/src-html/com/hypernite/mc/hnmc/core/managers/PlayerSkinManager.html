001package com.hypernite.mc.hnmc.core.managers;
002
003import org.bukkit.block.Block;
004import org.bukkit.block.BlockFace;
005import org.bukkit.inventory.ItemStack;
006
007import java.util.UUID;
008import java.util.concurrent.CompletableFuture;
009import java.util.function.Consumer;
010
011public interface PlayerSkinManager {
012
013    /**
014     * 生成頭顱
015     * @param uuid 玩家UUID
016     * @param callback 生成頭顱後的處理
017     */
018    void generateSkull(UUID uuid, Consumer<ItemStack> callback);
019
020    /**
021     * 生成頭顱
022     * @param uuid 玩家UUID
023     * @param name 玩家名稱
024     * @param callback 生成頭顱後的處理
025     */
026    void generateSkull(UUID uuid, String name, Consumer<ItemStack> callback);
027
028    /**
029     * @param b64  base64 Texture value
030     * @param head 頭顱
031     */
032    void setSkullMeta(String b64, ItemStack head);
033
034    /**
035     * 設置皮膚
036     * @param uuid 玩家UUID
037     * @param head 玩家名稱
038     * @param callback 設置皮膚後的頭顱
039     */
040    void setSkullMeta(UUID uuid, ItemStack head, Consumer<ItemStack> callback);
041
042    /**
043     * @param uuid 玩家UUID
044     * @param name 玩家名稱
045     * @param head 頭顱
046     * @param callback 設置皮膚後的頭顱
047     */
048    void setSkullMeta(UUID uuid, String name, ItemStack head, Consumer<ItemStack> callback);
049
050    /**
051     * @param uuid  玩家UUID
052     * @param block 方塊
053     * @param wall  是否貼墻
054     * @param face  頭顱方向
055     */
056    void setHeadBlock(UUID uuid, Block block, boolean wall, BlockFace face);
057
058    /**
059     * @param uuid   玩家UUID
060     * @param player 玩家名稱
061     * @param block  方塊
062     * @param wall   是否貼墻
063     * @param face   頭顱方向
064     */
065    void setHeadBlock(UUID uuid, String player, Block block, boolean wall, BlockFace face);
066
067    /**
068     * @param b64Value Base64 Texture Value
069     * @param block    方塊
070     * @param wall     是否貼墻
071     * @param face     頭顱方向
072     */
073    void setHeadBlock(String b64Value, Block block, boolean wall, BlockFace face);
074
075    /**
076     * @param uuid   玩家UUID
077     * @param player 玩家名稱
078     * @param block  方塊
079     */
080    void updateHeadBlock(UUID uuid, String player, Block block);
081
082    /**
083     * @param uuid  玩家UUID
084     * @param block 玩家名稱
085     */
086    void updateHeadBlock(UUID uuid, Block block);
087
088    /**
089     * @param newB64Value Base64 Texture Value
090     * @param block       方塊
091     */
092    void updateHeadBlock(String newB64Value, Block block);
093
094    /**
095     * @param uuid 玩家UUID
096     * @return Base64 Texture Value or IOException
097     */
098    CompletableFuture<String> getTextureValue(UUID uuid);
099
100}
