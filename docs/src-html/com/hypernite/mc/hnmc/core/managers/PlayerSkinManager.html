001package com.hypernite.mc.hnmc.core.managers;
002
003import org.bukkit.block.Block;
004import org.bukkit.block.BlockFace;
005import org.bukkit.inventory.ItemStack;
006
007import java.util.UUID;
008import java.util.concurrent.CompletableFuture;
009import java.util.function.Consumer;
010
011public interface PlayerSkinManager {
012
013    /**
014     * 生成頭顱
015     *
016     * @param uuid     玩家UUID
017     * @param callback 生成頭顱後的處理
018     */
019    void generateSkull(UUID uuid, Consumer<ItemStack> callback);
020
021    /**
022     * 生成頭顱
023     *
024     * @param uuid     玩家UUID
025     * @param name     玩家名稱
026     * @param callback 生成頭顱後的處理
027     */
028    void generateSkull(UUID uuid, String name, Consumer<ItemStack> callback);
029
030    /**
031     * @param b64  base64 Texture value
032     * @param head 頭顱
033     */
034    void setSkullMeta(String b64, ItemStack head);
035
036    /**
037     * 設置皮膚
038     *
039     * @param uuid     玩家UUID
040     * @param head     玩家名稱
041     * @param callback 設置皮膚後的頭顱
042     */
043    void setSkullMeta(UUID uuid, ItemStack head, Consumer<ItemStack> callback);
044
045    /**
046     * @param uuid     玩家UUID
047     * @param name     玩家名稱
048     * @param head     頭顱
049     * @param callback 設置皮膚後的頭顱
050     */
051    void setSkullMeta(UUID uuid, String name, ItemStack head, Consumer<ItemStack> callback);
052
053    /**
054     * @param uuid  玩家UUID
055     * @param block 方塊
056     * @param wall  是否貼墻
057     * @param face  頭顱方向
058     */
059    void setHeadBlock(UUID uuid, Block block, boolean wall, BlockFace face);
060
061    /**
062     * @param uuid   玩家UUID
063     * @param player 玩家名稱
064     * @param block  方塊
065     * @param wall   是否貼墻
066     * @param face   頭顱方向
067     */
068    void setHeadBlock(UUID uuid, String player, Block block, boolean wall, BlockFace face);
069
070    /**
071     * @param b64Value Base64 Texture Value
072     * @param block    方塊
073     * @param wall     是否貼墻
074     * @param face     頭顱方向
075     */
076    void setHeadBlock(String b64Value, Block block, boolean wall, BlockFace face);
077
078    /**
079     * @param uuid   玩家UUID
080     * @param player 玩家名稱
081     * @param block  方塊
082     */
083    void updateHeadBlock(UUID uuid, String player, Block block);
084
085    /**
086     * @param uuid  玩家UUID
087     * @param block 玩家名稱
088     */
089    void updateHeadBlock(UUID uuid, Block block);
090
091    /**
092     * @param newB64Value Base64 Texture Value
093     * @param block       方塊
094     */
095    void updateHeadBlock(String newB64Value, Block block);
096
097    /**
098     * @param uuid 玩家UUID
099     * @return Base64 Texture Value or IOException
100     */
101    CompletableFuture<String> getTextureValue(UUID uuid);
102
103}
