001package com.hypernite.mc.hnmc.core.managers;
002
003import org.bukkit.block.Block;
004import org.bukkit.block.BlockFace;
005import org.bukkit.inventory.ItemStack;
006
007import java.util.UUID;
008import java.util.concurrent.CompletableFuture;
009
010public interface PlayerSkinManager {
011
012    /**
013     * @param uuid 玩家UUID
014     * @return 有皮膚頭顱
015     */
016    ItemStack getSkull(UUID uuid);
017
018    /**
019     * @param uuid 玩家UUID
020     * @param name 玩家名稱
021     * @return 有皮膚的頭顱
022     */
023    ItemStack getSkull(UUID uuid, String name);
024
025    /**
026     * @param b64  base64 Texture value
027     * @param head 頭顱
028     */
029    void setSkullMeta(String b64, ItemStack head);
030
031    /**
032     * @param uuid 玩家UUID
033     * @param head 頭顱
034     */
035    void setSkullMeta(UUID uuid, ItemStack head);
036
037    /**
038     * @param uuid 玩家UUID
039     * @param name 玩家名稱
040     * @param head 頭顱
041     */
042    void setSkullMeta(UUID uuid, String name, ItemStack head);
043
044    /**
045     * @param uuid  玩家UUID
046     * @param block 方塊
047     * @param wall  是否貼墻
048     * @param face  頭顱方向
049     */
050    void setHeadBlock(UUID uuid, Block block, boolean wall, BlockFace face);
051
052    /**
053     * @param uuid   玩家UUID
054     * @param player 玩家名稱
055     * @param block  方塊
056     * @param wall   是否貼墻
057     * @param face   頭顱方向
058     */
059    void setHeadBlock(UUID uuid, String player, Block block, boolean wall, BlockFace face);
060
061    /**
062     * @param b64Value Base64 Texture Value
063     * @param block    方塊
064     * @param wall     是否貼墻
065     * @param face     頭顱方向
066     */
067    void setHeadBlock(String b64Value, Block block, boolean wall, BlockFace face);
068
069    /**
070     * @param uuid   玩家UUID
071     * @param player 玩家名稱
072     * @param block  方塊
073     */
074    void updateHeadBlock(UUID uuid, String player, Block block);
075
076    /**
077     * @param uuid  玩家UUID
078     * @param block 玩家名稱
079     */
080    void updateHeadBlock(UUID uuid, Block block);
081
082    /**
083     * @param newB64Value Base64 Texture Value
084     * @param block       方塊
085     */
086    void updateHeadBlock(String newB64Value, Block block);
087
088    /**
089     * @param uuid 玩家UUID
090     * @return Base64 Texture Value or IOException
091     */
092    CompletableFuture<String> getTextureValue(UUID uuid);
093
094}
