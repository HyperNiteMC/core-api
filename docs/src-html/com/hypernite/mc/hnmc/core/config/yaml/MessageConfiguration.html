001package com.hypernite.mc.hnmc.core.config.yaml;
002
003
004import com.hypernite.mc.hnmc.core.config.MessageGetter;
005
006import java.util.List;
007
008/**
009 * 用於創建 訊息類別 config 時繼承
010 */
011public abstract class MessageConfiguration extends Configuration {
012
013    private MessageGetter messageGetter;
014
015    /**
016     * 若果 訊息類別 config 本身沒有 @Prefix, 返回的將是 null
017     *
018     * @return 前綴
019     */
020    public String getPrefix() {
021        return messageGetter.getPrefix();
022    }
023
024    /**
025     * @param path 路徑
026     * @return 獲取訊息
027     */
028    public String get(String path) {
029        return messageGetter.get(path);
030    }
031
032    /**
033     * @param path 路徑
034     * @return 獲取無前綴訊息
035     */
036    public String getPure(String path) {
037        return messageGetter.getPure(path);
038    }
039
040    /**
041     * @param path 路徑
042     * @return 獲取無前綴訊息列表
043     */
044    public List<String> getPureList(String path) {
045        return messageGetter.getPureList(path);
046    }
047
048    /**
049     * @param path 路徑
050     * @return 獲取訊息列表
051     */
052    public List<String> getList(String path) {
053        return messageGetter.getList(path);
054    }
055}
