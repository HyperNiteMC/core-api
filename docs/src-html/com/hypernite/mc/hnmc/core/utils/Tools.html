001package com.hypernite.mc.hnmc.core.utils;
002
003/**
004 * Normal Tools
005 */
006public class Tools {
007
008    public static int randomWithRange(int min, int max) {
009        int range = (max - min) + 1;
010        return (int) (Math.random() * range) + min;
011    }
012
013    public static String toLongString(String[] args, int startFrom) {
014        StringBuilder arg = new StringBuilder(startFrom);
015        for (int i = startFrom; i < args.length; i++) {
016            String str = args[i];
017            arg.append(' ').append(str);
018        }
019        return arg.toString();
020    }
021
022}
