003import org.bukkit.Bukkit;
004import org.bukkit.entity.Player;

006import java.lang.reflect.Constructor;

017/**
018 * Java Reflection Tools
019 */
020public class ReflectionUtil {
021
022    /*
023     * The server version string to location NMS & OBC classes
024     */
025    private static String versionString;
026
027    /*
028     * Cache of NMS classes that we've searched for
029     */
030    private static Map<String, Class<?>> loadedNMSClasses = new HashMap<String, Class<?>>();
031
032    /*
033     * Cache of OBS classes that we've searched for
034     */
035    private static Map<String, Class<?>> loadedOBCClasses = new HashMap<String, Class<?>>();
036
037    /*
038     * Cache of methods that we've found in particular classes
039     */
040    private static Map<Class<?>, Map<String, Method>> loadedMethods = new HashMap<Class<?>, Map<String, Method>>();
041
042    /*
043     * Cache of fields that we've found in particular classes
044     */
045    private static Map<Class<?>, Map<String, Field>> loadedFields = new HashMap<Class<?>, Map<String, Field>>();
046
047    /**
048     * Gets the version string for NMS and OBC class paths
049     *
050     * @return The version string of OBC and NMS packages
051     */
052    public static String getVersion() {
053        if (versionString == null) {
054            String name = Bukkit.getServer().getClass().getPackage().getName();
055            versionString = name.substring(name.lastIndexOf('.') + 1) + ".";
056        }
057
058        return versionString;
059    }
060
061    /**
062     * Get an NMS Class
063     *
064     * @param nmsClassName The name of the class
065     * @return The class
066     */
067    public static Optional<Class<?>> getNMSClass(String nmsClassName) {
068        if (loadedNMSClasses.containsKey(nmsClassName)) {
069            return Optional.ofNullable(loadedNMSClasses.get(nmsClassName));
070        }
071        String clazzName = "net.minecraft.server." + getVersion() + nmsClassName;
072        Class<?> clazz;
073
074        try {
075            clazz = Class.forName(clazzName);
076        } catch (Throwable t) {
077            t.printStackTrace();
078            return Optional.ofNullable(loadedNMSClasses.put(nmsClassName, null));
079        }
080
081        loadedNMSClasses.put(nmsClassName, clazz);
082        return Optional.ofNullable(clazz);
083    }
084
085    /**
086     * Get a class from the org.bukkit.craftbukkit package
087     *
088     * @param obcClassName the path to the class
089     * @return the found class at the specified path
090     */
091    public synchronized static Optional<Class<?>> getOBCClass(String obcClassName) {
092        if (loadedOBCClasses.containsKey(obcClassName)) {
093            return Optional.of(loadedOBCClasses.get(obcClassName));
094        }
095
096        String clazzName = "org.bukkit.craftbukkit." + getVersion() + obcClassName;
097        Class<?> clazz;
098
099        try {
100            clazz = Class.forName(clazzName);
101        } catch (Throwable t) {
102            t.printStackTrace();
103            loadedOBCClasses.put(obcClassName, null);
104            return Optional.empty();
105        }
106
107        loadedOBCClasses.put(obcClassName, clazz);
108        return Optional.ofNullable(clazz);
109    }
110
111    /**
112     * Get a Bukkit {@link Player} players NMS playerConnection object
113     *
114     * @param player The player
115     * @return The players connection
116     */
117    public static Optional<Object> getConnection(Player player) {
118        Optional<Method> getHandleMethod = getMethod(player.getClass(), "getHandle");
119
120        if (getHandleMethod.isPresent()) {
121            try {
122                Object nmsPlayer = getHandleMethod.get().invoke(player);
123                Field playerConField = getField(nmsPlayer.getClass(), "playerConnection").orElseThrow();
124                return Optional.ofNullable(playerConField.get(nmsPlayer));
125            } catch (Exception e) {
126                e.printStackTrace();
127            }
128        }
129        return Optional.empty();
130    }
131
132    /**
133     * Get a classes constructor
134     *
135     * @param clazz  The constructor class
136     * @param params The parameters in the constructor
137     * @return The constructor object
138     */
139    public static Optional<Constructor<?>> getConstructor(Class<?> clazz, Class<?>... params) {
140        try {
141            return Optional.ofNullable(clazz.getConstructor(params));
142        } catch (NoSuchMethodException e) {
143            return Optional.empty();
144        }
145    }
146
147    /**
148     * Get a method from a class that has the specific paramaters
149     *
150     * @param clazz      The class we are searching
151     * @param methodName The name of the method
152     * @return The method with appropriate paramaters
153     */
154    public static Optional<Method> getMethod(Class<?> clazz, String methodName) {
155        if (!loadedMethods.containsKey(clazz)) {
156            loadedMethods.put(clazz, new HashMap<String, Method>());
157        }
158
159        Map<String, Method> methods = loadedMethods.get(clazz);
160
161        if (methods.containsKey(methodName)) {
162            return Optional.ofNullable(methods.get(methodName));
163        }
164
165        try {
166            Method method = clazz.getMethod(methodName);
167            methods.put(methodName, method);
168            loadedMethods.put(clazz, methods);
169            return Optional.ofNullable(method);
170        } catch (Exception e) {
171            e.printStackTrace();
172            methods.put(methodName, null);
173            loadedMethods.put(clazz, methods);
174            return Optional.empty();
175        }
176    }
177
178    /**
179     * Get a method from a class that has the specific paramaters
180     *
181     * @param clazz      The class we are searching
182     * @param methodName The name of the method
183     * @param params     Any parameters that the method has
184     * @return The method with appropriate paramaters
185     */
186    public static Optional<Method> getMethod(Class<?> clazz, String methodName, Class<?>... params) {
187        if (!loadedMethods.containsKey(clazz)) {
188            loadedMethods.put(clazz, new HashMap<String, Method>());
189        }
190
191        Map<String, Method> methods = loadedMethods.get(clazz);
192
193        if (methods.containsKey(methodName)) {
194            return Optional.ofNullable(methods.get(methodName));
195        }
196
197        try {
198            Method method = clazz.getMethod(methodName, params);
199            methods.put(methodName, method);
200            loadedMethods.put(clazz, methods);
201            return Optional.ofNullable(method);
202        } catch (Exception e) {
203            e.printStackTrace();
204            methods.put(methodName, null);
205            loadedMethods.put(clazz, methods);
206            return Optional.empty();
207        }
208    }
209
210    /**
211     * Get a field with a particular name from a class
212     *
213     * @param clazz     The class
214     * @param fieldName The name of the field
215     * @return The field object
216     */
217    public static Optional<Field> getField(Class<?> clazz, String fieldName) {
218        if (!loadedFields.containsKey(clazz)) {
219            loadedFields.put(clazz, new HashMap<String, Field>());
220        }
221
222        Map<String, Field> fields = loadedFields.get(clazz);
223
224        if (fields.containsKey(fieldName)) {
225            return Optional.ofNullable(fields.get(fieldName));
226        }
227
228        try {
229            Field field = clazz.getField(fieldName);
230            fields.put(fieldName, field);
231            loadedFields.put(clazz, fields);
232            return Optional.ofNullable(field);
233        } catch (Exception e) {
234            e.printStackTrace();
235            fields.put(fieldName, null);
236            loadedFields.put(clazz, fields);
237            return Optional.empty();
238        }
239    }
