001package com.hypernite.mc.hnmc.core.utils.converters;
002
003import org.bukkit.Bukkit;
004import org.bukkit.inventory.Inventory;
005import org.bukkit.inventory.ItemStack;
006import org.bukkit.inventory.PlayerInventory;
007import org.bukkit.util.io.BukkitObjectInputStream;
008import org.bukkit.util.io.BukkitObjectOutputStream;
009import org.yaml.snakeyaml.external.biz.base64Coder.Base64Coder;
010
011import java.io.ByteArrayInputStream;
012import java.io.ByteArrayOutputStream;
013import java.io.IOException;
014
015/**
016 * 背包 Base64 Value 輸入/輸出
017 */
018public class InvConvert {
019
020    public static String[] playerInventoryToBase64(PlayerInventory playerInventory) {
021        //get the main content part, this doesn't return the armor
022        String content = toBase64(playerInventory);
023        String armor = ItemConvert.itemStackArrayToBase64(playerInventory.getArmorContents());
024        return new String[]{content, armor};
025    }
026
027
028    public static String toBase64(Inventory inventory) {
029        try {
030            ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
031            BukkitObjectOutputStream dataOutput = new BukkitObjectOutputStream(outputStream);
032
033            // Write the size of the inventory
034            dataOutput.writeInt(inventory.getSize());
035
036            // Save every element in the list
037            for (int i = 0; i < inventory.getSize(); i++) {
038                dataOutput.writeObject(inventory.getItem(i));
039            }
040
041            // Serialize that array
042            dataOutput.close();
043            return Base64Coder.encodeLines(outputStream.toByteArray());
044        } catch (Exception e) {
045            throw new IllegalStateException("Unable to save item stacks.", e);
046        }
047    }
048
049    public static Inventory fromBase64(String data) {
050        try {
051            ByteArrayInputStream inputStream = new ByteArrayInputStream(Base64Coder.decodeLines(data));
052            BukkitObjectInputStream dataInput = new BukkitObjectInputStream(inputStream);
053            Inventory inventory = Bukkit.getServer().createInventory(null, dataInput.readInt());
054
055            // Read the serialized inventory
056            for (int i = 0; i < inventory.getSize(); i++) {
057                inventory.setItem(i, (ItemStack) dataInput.readObject());
058            }
059
060            dataInput.close();
061            return inventory;
062        } catch (ClassNotFoundException | IOException e) {
063            e.printStackTrace();
064            return null;
065        }
066    }
067}
