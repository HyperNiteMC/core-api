001package com.hypernite.mc.hnmc.core.misc.permission;
002
003import org.bukkit.command.CommandSender;
004
005import javax.annotation.Nonnull;
006import java.util.ArrayList;
007import java.util.List;
008
009/**
010 * 權限 API
011 */
012public class Perm {
013    public static final String ADMIN = "hypernite.admin";
014    public static final String HELPER = "hypernite.helper";
015    public static final String MOD = "hypernite.mod";
016    public static final String BUILDER = "hypernite.builder";
017    public static final String DONOR = "hypernite.donor";
018    public static final String OWNER = "*";
019
020    public static boolean hasPermission(CommandSender sender, String permission) {
021        boolean reuslt = false;
022        if (permission == null) return true;
023        for (String Pnode : getPermHierarchy(permission)) {
024            reuslt = reuslt || sender.hasPermission(Pnode);
025        }
026        return reuslt;
027    }
028
029    public static List<String> getPermHierarchy(@Nonnull String string) {
030        String[] node = string.split("\\.");
031        List<String> permissionNode = new ArrayList<>();
032        StringBuilder stringBuilder = new StringBuilder();
033        for (int i = 0; i < node.length; i++) {
034            stringBuilder.append(node[i]);
035            permissionNode.add(stringBuilder.toString());
036            if (i != node.length - 1) stringBuilder.append(".");
037        }
038        return permissionNode;
039    }
040}
