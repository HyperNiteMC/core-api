001package com.hypernite.mc.hnmc.core.misc.world;
002
003import org.bukkit.Location;
004
005public class WorldProperties {
006    private boolean pvp;
007    private boolean pve;
008    private Location spawn;
009    private boolean vulnerable;
010    private boolean autoLoad;
011
012    public WorldProperties(boolean pvp, boolean pve, Location spawn, boolean vulnerable, boolean autoLoad) {
013        this.pvp = pvp;
014        this.pve = pve;
015        this.spawn = spawn;
016        this.vulnerable = vulnerable;
017        this.autoLoad = autoLoad;
018    }
019
020    public WorldProperties(Location spawn) {
021        this.spawn = spawn;
022    }
023
024    public boolean isPvp() {
025        return pvp;
026    }
027
028    public void setPvp(boolean pvp) {
029        this.pvp = pvp;
030    }
031
032    public boolean isPve() {
033        return pve;
034    }
035
036    public void setPve(boolean pve) {
037        this.pve = pve;
038    }
039
040    public Location getSpawn() {
041        return spawn;
042    }
043
044    public void setSpawn(Location spawn) {
045        this.spawn = spawn;
046    }
047
048    public boolean isVulnerable() {
049        return vulnerable;
050    }
051
052    public void setVulnerable(boolean vulnerable) {
053        this.vulnerable = vulnerable;
054    }
055
056    public boolean isAutoLoad() {
057        return autoLoad;
058    }
059
060    public void setAutoLoad(boolean autoLoad) {
061        this.autoLoad = autoLoad;
062    }
063}
