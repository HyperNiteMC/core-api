001package com.hypernite.mc.hnmc.core.builders;
002
003import com.hypernite.mc.hnmc.core.main.HyperNiteMC;
004import com.hypernite.mc.hnmc.core.managers.builder.AbstractItemStackBuilder;
005import org.bukkit.Material;
006import org.bukkit.enchantments.Enchantment;
007import org.bukkit.event.inventory.InventoryClickEvent;
008import org.bukkit.event.player.PlayerInteractEvent;
009import org.bukkit.inventory.Inventory;
010import org.bukkit.inventory.ItemFlag;
011import org.bukkit.inventory.ItemStack;
012
013import java.util.List;
014import java.util.Map;
015import java.util.UUID;
016import java.util.function.Consumer;
017import java.util.function.Supplier;
018
019/**
020 * @see com.hypernite.mc.hnmc.core.managers.builder.AbstractItemStackBuilder
021 */
022public class ItemStackBuilder implements AbstractItemStackBuilder {
023
024    private final AbstractItemStackBuilder delegate;
025
026    /**
027     * @param m 物品類型
028     */
029    public ItemStackBuilder(Material m) {
030        delegate = HyperNiteMC.getAPI().getFactory().getBuilder().getItemStackBuilder(m);
031    }
032
033    public ItemStackBuilder() {
034        delegate = HyperNiteMC.getAPI().getFactory().getBuilder().getItemStackBuilder();
035    }
036
037    /**
038     * @param item 現有物品
039     */
040    public ItemStackBuilder(ItemStack item) {
041        delegate = HyperNiteMC.getAPI().getFactory().getBuilder().getItemStackBuilder(item);
042    }
043
044    @Override
045    public AbstractItemStackBuilder material(Material m) {
046        return delegate.material(m);
047    }
048
049    @Override
050    public AbstractItemStackBuilder durability(int dur) {
051        return delegate.durability(dur);
052    }
053
054    @Override
055    public AbstractItemStackBuilder enchant(Enchantment enchantment, int level) {
056        return delegate.enchant(enchantment, level);
057    }
058
059    @Override
060    public AbstractItemStackBuilder enchant(Map<Enchantment, Integer> enchantmentMap) {
061        return delegate.enchant(enchantmentMap);
062    }
063
064    @Override
065    public AbstractItemStackBuilder openGui(Supplier<Inventory> inventorySupplier) {
066        return delegate.openGui(inventorySupplier);
067    }
068
069    @Override
070    public AbstractItemStackBuilder stack(int s) {
071        return delegate.stack(s);
072    }
073
074    @Override
075    public AbstractItemStackBuilder displayName(String name) {
076        return delegate.displayName(name);
077    }
078
079    @Override
080    public AbstractItemStackBuilder lore(String... lore) {
081        return delegate.lore(lore);
082    }
083
084    @Override
085    public AbstractItemStackBuilder lore(List<String> lore) {
086        return delegate.lore(lore);
087    }
088
089    @Override
090    public AbstractItemStackBuilder unbreakable(boolean unbreakable) {
091        return delegate.unbreakable(unbreakable);
092    }
093
094    @Override
095    public AbstractItemStackBuilder onClick(Consumer<InventoryClickEvent> action) {
096        return delegate.onClick(action);
097    }
098
099    @Override
100    public AbstractItemStackBuilder onInteract(Consumer<PlayerInteractEvent> action) {
101        return delegate.onInteract(action);
102    }
103
104    @Override
105    public AbstractItemStackBuilder head(UUID uuid) {
106        return delegate.head(uuid);
107    }
108
109    @Override
110    public AbstractItemStackBuilder head(UUID uuid, String player) {
111        return delegate.head(uuid, player);
112    }
113
114    @Override
115    public AbstractItemStackBuilder modelData(int data) {
116        return delegate.modelData(data);
117    }
118
119    @Override
120    public AbstractItemStackBuilder itemFlags(ItemFlag... itemFlags) {
121        return delegate.itemFlags(itemFlags);
122    }
123
124    @Override
125    public ItemStack build() {
126        return delegate.build();
127    }
128}
