001package com.hypernite.mc.hnmc.core.builders;
002
003import com.hypernite.mc.hnmc.core.main.HyperNiteMC;
004import com.hypernite.mc.hnmc.core.managers.builder.AbstractInventoryBuilder;
005import org.bukkit.inventory.Inventory;
006import org.bukkit.inventory.ItemStack;
007
008/**
009 * @see com.hypernite.mc.hnmc.core.managers.builder.AbstractInventoryBuilder
010 */
011public class InventoryBuilder implements AbstractInventoryBuilder {
012    /**
013     * 第一行
014     */
015    public static final int ONE_ROW = AbstractInventoryBuilder.ONE_ROW;
016    /**
017     * 中央位置
018     */
019    public static final int CENTER = AbstractInventoryBuilder.CENTER;
020    /**
021     * 初始位置
022     */
023    public static final int START = AbstractInventoryBuilder.START;
024    /**
025     * 尾部位置
026     */
027    public static final int END = AbstractInventoryBuilder.END;
028
029
030    private final AbstractInventoryBuilder delegate;
031
032    /**
033     * @param row   此背包界面最大行數
034     * @param title 標題
035     */
036    public InventoryBuilder(int row, String title) {
037        this.delegate = HyperNiteMC.getAPI().getFactory().getBuilder().getInventoryBuilder(row, title);
038    }
039
040
041    @Override
042    public AbstractInventoryBuilder item(ItemStack item) {
043        return delegate.item(item);
044    }
045
046
047    @Override
048    public AbstractInventoryBuilder item(int slot, ItemStack item) {
049        return delegate.item(slot, item);
050    }
051
052
053    @Override
054    public AbstractInventoryBuilder item(int row, int slot, ItemStack item) {
055        return delegate.item(row, slot, item);
056    }
057
058
059    @Override
060    public AbstractInventoryBuilder center(ItemStack item) {
061        return delegate.center(item);
062    }
063
064
065    @Override
066    public AbstractInventoryBuilder ring(ItemStack item) {
067        return delegate.ring(item);
068    }
069
070
071    @Override
072    public AbstractInventoryBuilder fillRow(int row, ItemStack item) {
073        return delegate.fillRow(row, item);
074    }
075
076    @Override
077    public Inventory build() {
078        return delegate.build();
079    }
080}
